document.addEventListener('DOMContentLoaded', function() {
    const gallery = document.querySelector('.gallery');

    // Данные для изображений (замените своими)
    const images = [
        { src: 'images/istockphoto-500466008-2048x2048.webp', alt: 'Блюдо 1' },
          { src: 'images/istockphoto-500466008-2048x2048.webp', alt: 'Блюдо 1' },
       { src: 'images/istockphoto-500466008-2048x2048.webp', alt: 'Блюдо 1' },,
     { src: 'images/istockphoto-500466008-2048x2048.webp', alt: 'Блюдо 1' },
     { src: 'images/istockphoto-500466008-2048x2048.webp', alt: 'Блюдо 1' },
     { src: 'images/istockphoto-500466008-2048x2048.webp', alt: 'Блюдо 1' },
        // ... добавьте свои изображения
    ];

    // Функция для создания элемента изображения
    function createImageElement(image) {
        const imgLink = document.createElement('a');
        imgLink.href = image.src; 
        imgLink.target = "_blank"; 

        const img = document.createElement('img');
        img.src = image.src;
        img.alt = image.alt;
        img.loading = "lazy"; // Ленивая загрузка

        // Добавляем img в imgLink
        imgLink.appendChild(img);
        // Добавляем imgLink в галерею
        gallery.appendChild(imgLink);
    }

    // Загрузка изображений в галерею
    images.forEach(createImageElement);

    // Липкий header
    window.addEventListener('scroll', () => {
        const header = document.getElementById('header');
        if (window.scrollY > 0) {
            header.classList.add('sticky');
        } else {
            header.classList.remove('sticky');
        }
    });
});

#send_email.php--это файл в папке
<?php
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Получаем данные формы
    $name = htmlspecialchars($_POST['name']);
    $email = htmlspecialchars($_POST['email']);
    $message = htmlspecialchars($_POST['message']);

    // Email, на который будет отправлено сообщение
    $to = 'dimash.abenov@inbox.ru';

    // Тема письма
    $subject = 'Новое сообщение с сайта';

    // Формируем тело письма
    $body = "Имя: $name\n";
    $body .= "Email: $email\n";
    $body .= "Сообщение: \n$message\n";

    // Заголовки письма
    $headers = "From: $email\r\n";
    $headers .= "Reply-To: $email\r\n";
    $headers .= "Content-Type: text/plain; charset=utf-8\r\n";

    // Печать отладочной информации
    echo "<pre>";
    echo "Отправка сообщения на: $to\n";
    echo "Тема: $subject\n";
    echo "Тело сообщения:\n$body\n";
    echo "Заголовки:\n$headers\n";
    echo "</pre>";

    // Отправляем письмо
    if (mail($to, $subject, $body, $headers)) {
        echo 'Спасибо! Ваше сообщение отправлено.';
    } else {
        echo 'Извините, произошла ошибка при отправке вашего сообщения.';
    }
} else {
    echo 'Неверный запрос.';
}
?>
